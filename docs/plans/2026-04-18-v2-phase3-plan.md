```rust
// traits.rs
pub enum ToolLayer { L0, L1, L2 }
impl ToolLayer { pub fn prefix(&self) -> &'static str { match self { Self::L0 => "l0", Self::L1 => "l1", Self::L2 => "l2" } } }
pub trait Tool: Send + Sync {
    fn name(&self) -> &str;
    fn layer(&self) -> ToolLayer { ToolLayer::L1 }  // default
    async fn execute(&self, params: Value) -> Result<ToolResult>;
}
```

```rust
// mod.rs
pub struct ToolRegistry {
    tools: HashMap<(ToolLayer, String), Arc<dyn Tool>>,
}
impl ToolRegistry {
    pub fn register_layered(&mut self, layer: ToolLayer, tool: Arc<dyn Tool>) { ... }
    pub fn resolve(&self, full_name: &str) -> Option<Arc<dyn Tool>> { ... }  // 解析 "l1:bash.execute"
    pub fn resolve_with_fallback(&self, name: &str) -> Option<Arc<dyn Tool>> {
        self.tools.get(&(ToolLayer::L2, name.into()))
            .or_else(|| self.tools.get(&(ToolLayer::L1, name.into())))
            .or_else(|| self.tools.get(&(ToolLayer::L0, name.into())))
            .cloned()
    }
}
```
